*PARÂMETROS
.param length = 32nm
.param width = 64nm

* VALIDAÇÃO CIRCUITO DE ESCRITA SRAM 6T
.include 32nm_HP.pm

* DECLARAÇÃO DAS FONTES DE TENSÃO
Vvdd vdd gnd 1

* DECLARAÇÃO DO CIRCUITO
* NOMENCLATURA: nome source gate dreno bulk(vdd/gnd) TIPO_PMOS_OU_NMOS w=  l=

* INVERSOR DO BIT
M1_bit vdd bit bitb vdd PMOS w = width l = length
M2_bit bitb bit gnd gnd NMOS w = width l = length

* INVERSOR DO WRITE ENABLE (WE)
M1_we vdd we web vdd PMOS w = width l = length
M2_we web we gnd gnd NMOS w = width l = length

* ESCRITA EM BL
M1 gnd bitb out_bl gnd NMOS w = width l = length
M2 out_bl we bl gnd NMOS w = width l = length
M3 vdd bitb out_bl vdd PMOS w = width l = length
M4 out_bl web bl vdd PMOS w = width l = length

* ESCRITA EM BLB
M5 gnd bit out_blb gnd NMOS w = width l = length
M6 out_blb we blb gnd NMOS w = width l = length
M7 vdd bit out_blb vdd PMOS w = width l = length
M8 out_blb web blb vdd PMOS w = width l = length

.end
